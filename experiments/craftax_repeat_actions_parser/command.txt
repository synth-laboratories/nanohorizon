def baseline_sanitize(values: list[object]) -> list[str]:
    sanitized: list[str] = []
    for value in values:
        raw = str(value).strip().lower()
        if not raw:
            continue
        if raw in rollout_module.ACTION_NAME_TO_INDEX and raw not in sanitized:
            sanitized.append(raw)
            continue
        for token in __import__("re").findall(r"[a-z_]+", raw):
            if token in rollout_module.ACTION_NAME_TO_INDEX and token not in sanitized:
                sanitized.append(token)
    return sanitized


class FakeRunner:
    def __init__(self, seed: int) -> None:
        self.seed = seed
        self.state = {"steps": 0, "seed": seed}
        self.action_history: list[int] = []

    def reset(self):
        self.state = {"steps": 0, "seed": self.seed}
        self.action_history = []
        return SimpleNamespace(
            done=False,
            render=RenderBundle(mode=RenderMode.TEXT, text=f"obs seed={self.seed} steps=0", pixels=None),
            reward=0.0,
            info={},
            step_index=0,
            episode_index=0,
        )

    def step_many(self, actions):
        outputs = []
        for action in actions:
            self.action_history.append(int(action))
            self.state["steps"] += 1
            outputs.append(
                SimpleNamespace(
                    done=False,
                    reward=1.0,
                    render=RenderBundle(
                        mode=RenderMode.TEXT,
                        text=f"obs seed={self.seed} steps={self.state['steps']}",
                        pixels=None,
                    ),
                )
            )
        return outputs
